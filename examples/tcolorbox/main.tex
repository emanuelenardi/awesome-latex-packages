\DeclareRobustCommand{\ulseries}{\fontseries{ul}\selectfont}% Ultralight --> Thin

\DeclareRobustCommand{\sbseries}{\fontseries{sb}\selectfont}% Semibold

\DeclareRobustCommand{\xbseries}{\fontseries{xb}\selectfont}% Black

\DeclareTextFontCommand{\textsb}{\sbseries}					% Semibold
